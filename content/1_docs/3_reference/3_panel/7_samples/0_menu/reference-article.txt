```yaml "/site/blueprints/fields/dishes.yml"
label: Dishes
type: structure
fields:
  dish:
    label: Dish
    type: text
    width: 1/3
  description:
    label: Description
    type: text
    width: 1/3
  price:
```

## Example controller and template

### Controller

```php "site/controllers/menu.php
<?php
return function () {
    $categories = [
        'Starters',
        'Pasta',
        'Meat',
        'Desert'
    ];
    return [
        'categories' => $categories
    ];
};
```

### Template

```php "site/templates/menu.php
<?php snippet('header') ?>

<main class="restaurant-menu">
  <header class="intro">

    <div class="intro-text text">
      <?= $page->intro()->kt() ?></p>
```
