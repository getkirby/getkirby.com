----
phpTemplate:

```php
<?php snippet('header') ?>

<article class="album">
  <h1><?= $page->title() ?></h1>
  <figure class="cover">
    <?= $cover->resize(800, 600) ?>
  </figure>
  <div class="text">
    <?= $page->text()->kirbytext() ?>
  </div>
  <ul class="gallery">
    <?php foreach ($gallery as $image): ?>
    <li><?= $image->crop(300) ?></li>
    <?php endforeach ?>
  </ul>
</article>

<?php snippet('footer') ?>
```

----
twigTemplate:

```php
{% extends 'layout.twig' %}

{% block content %}
<article class="album">
  <h1>{{ page.title }}</h1>
  <figure class="cover">
    {{ cover.resize(800, 600) }}
  </figure>
  <div class="text">
    {{ page.text.kirbytext | raw }}
  </div>
  <ul class="gallery">
    {% for image in gallery %}
    <li>{{ image.crop(300) }}</li>
    {% endfor %}
  </ul>
</article>
{% endblock %}
```

----

bladeTemplate:

```php
@extends('layouts.default')

@section('content')
<article class="album">
  <h1>{{ $page->title() }}</h1>
  <figure class="cover">
    {!! $cover->resize(800, 600) !!}
  </figure>
  <div class="text">
    {!! $page->text()->kirbytext() !!}
  </div>
  <ul class="gallery">
    @foreach($gallery as $image)
    <li>{!! $image->crop(300) !!}</li>
    @endforeach
  </ul>
</article>
@endsection
```
