Templates:

```php
<?php snippet('header') ?>

<article class="album">
  <h1><?= $page->title() ?></h1>
  <figure class="cover">
    <?= $cover->resize(800, 600) ?>
  </figure>
  <div class="text">
    <?= $page->text()->kirbytext() ?>
  </div>
  <ul class="gallery">
    <?php foreach ($gallery as $image): ?>
    <li><?= $image->crop(300) ?></li>
    <?php endforeach ?>
  </ul>
</article>

<?php snippet('footer') ?>
```

Twig:

```html "/site/templates/article.twig"

  </div>
  <ul class="gallery">
    {% for image in page.images() %}

    {% endfor %}
  </ul>
</article>
```
